242  {
243 
244  ACE_Write_Guard<ACE_Thread_Mutex> guard(this->asset_guard);
245 
246  this->argc = argc;

249 
250

271  ACE_TRY_CHECK;
272  }
273  ACE_CATCH (CosNaming::NamingContext::AlreadyBound, ex)
274  {
275  ACE_PRINT_EXCEPTION (ex, "name service already bound :");
276  this->disconnect();
277  ACE_RE_THROW;
278  }
279  ACE_CATCH (CORBA::SystemException, sysex)
280  {
281  ACE_PRINT_EXCEPTION (sysex,
282  "System Exception initializing EC_Adapter:");
283  this->disconnect();
284  ACE_RE_THROW;

286  ACE_CATCH (CORBA::UserException, userex)
287  {
288  ACE_PRINT_EXCEPTION (userex,
289  "User Exception initializing EC_Adapter:");
290  this->disconnect();
291  ACE_RE_THROW;
292  }
293 
294  ACE_CATCHANY
295  {
296  ACE_PRINT_EXCEPTION (ACE_ANY_EXCEPTION,
297  "Caught exception initializing EC_Adapter:");
298  this->disconnect();
299  ACE_RE_THROW;
300  }
301  ACE_ENDTRY;
302  }
303 
304  int slot_num;
305 
306  this->add_supplier(slot_num, event_channel_name_p,
307  this->my_id, this->send_event_type) ;
308  this->add_consumer(slot_num, event_channel_name_p, this->get_event_type) ;
309  return;
310 }

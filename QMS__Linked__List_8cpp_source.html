262 QMS_Linked_List<NodeType, KeyType, Lock>::delete_element(QMS_Node<NodeType, KeyType, Lock> *delete_p)
263 {
264  QMS_Trace ("QMS_Linked_List<NodeType, KeyType, Lock>::delete_element",

268 
269  ACE_Write_Guard<Lock> guard(lock_);
270  // refuse to delete the head node
271  if(delete_p == this->head_p_)

273  return 0;

275 
276  // Save the data

278  KeyType* key_p = delete_p->get_key();
279  delete key_p;
280 
281  delete_p->prev_p_->next_p_ = delete_p->next_p_;

283  delete_p->next_p_ = delete_p->prev_p_
284  = 0;

286  delete_p = 0;
287 
288  this->size_--;
289  return data_p;

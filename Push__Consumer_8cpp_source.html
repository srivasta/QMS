443 STDC::QMS::Push_Consumer::connect (CORBA::Environment &ACE_TRY_ENV)
444  ACE_THROW_SPEC ((CORBA::SystemException))
445 {
446  QMS_Trace ("STDC::QMS::Push_Consumer::connect", __LINE__, __FILE__);
447  DEBUG0(6, "(%P|%t) New QMS Consumer connecting\n");
448  ACE_Write_Guard<ACE_Thread_Mutex> guard(this->asset_guard);
449 
450  ACE_TRY
451  {
452  if(this->Is_Scheduled == false)
453  {
454  this->default_schedule();
455  } /* end of if(this->Is_Scheduled == false) */
456 
457  // We need to get a supplier admin object from the event channel
458  // The canonical protocol to connect to the EC
459  if (CORBA::is_nil (this->_consumer_admin.in ())) {
460  RtecEventChannelAdmin::EventChannel_var event_channel_var;
461  DEBUG1(4, "DEBUG: Connecting to ec %s", this->ec_name_p);
462  DEBUG1(4, "DEBUG: Getting event type %ld", this->_get_event_type);
463  this->handles.get_event_channel(this->ec_name_p, event_channel_var,
464  ACE_TRY_ENV);
465  RtecEventChannelAdmin::EventChannel_ptr event_channel =
466  event_channel_var.in();
467 
468  this->_consumer_admin = event_channel->for_consumers (ACE_TRY_ENV);
469  ACE_TRY_CHECK;
470  DEBUG0 (6, "Consumer gettingt consumer admin\n");
471  int count = 0;
472  while (CORBA::is_nil (this->_consumer_admin.in ()))
473  {
474  ACE_Time_Value sleep_time (QMS_DEFAULT_retry_sleep_interval, 0);
475  ACE_ERROR ((LM_ERROR,
476  "(%P|%t) Unable to get the admin object.\n"));
477  if(count >= 4)
478  {
479  throw std::runtime_error::runtime_error("Unable to get the admin object.\n");
480  } /* end of if(count >= 4) */
481 
482  ACE_OS::sleep (sleep_time);
483  DEBUG0 (6, "retrying getting admin object\n");
484  this->_consumer_admin =
485  this->handles.get_ec_handle()->for_consumers (ACE_TRY_ENV);
486  ACE_TRY_CHECK;
487  count++;
488  }
489  DEBUG0 (6, "Got admin object\n");
490  }
491  else
492  {
493  DEBUG0 (6, "Already had a admin object\n");
494  }
495 
496 
497  if (CORBA::is_nil (this->_supplier_proxy.in ()))
498  {
499  // Get a push supplier proxy from the event channel
500  DEBUG0 (6, "getting supplier proxy\n");
501  this->_supplier_proxy =
502  this->_consumer_admin->obtain_push_supplier (ACE_TRY_ENV);
503  ACE_TRY_CHECK;
504  int count = 0;
505  while (CORBA::is_nil (this->_supplier_proxy.in ())){
506  ACE_Time_Value sleep_time (QMS_DEFAULT_retry_sleep_interval, 0);
507  ACE_ERROR((LM_ERROR,"(%P|%t) Unable to get the supplier proxy\n"));
508  if(count >= 4)
509  {
510  throw std::runtime_error::runtime_error("Unable to get the admin object.\n");
511  } /* end of if(count >= 4) */
512  ACE_OS::sleep (sleep_time);
513  DEBUG0 (6, "retrying getting supplier proxy\n");
514  this->_supplier_proxy =
515  this->_consumer_admin->obtain_push_supplier (ACE_TRY_ENV);
516  ACE_TRY_CHECK;
517 
518  count++;
519  }
520  DEBUG0 (6, "Got supplier proxy\n");
521  }
522  else
523  {
524  DEBUG0 (6, "Already had a supplier proxy\n");
525  }
526 
527  if (this->connected_to_proxy == false){
528  // set ourselves up as consumers
529  RtecEventComm::PushConsumer_var consumer =
530  this->_this (ACE_TRY_ENV);
531  ACE_TRY_CHECK;
532 
533  // register with the event channel
534  DEBUG0 (6, "connecting consumer\n");
535  this->_supplier_proxy->connect_push_consumer

539  DEBUG0 (6, "consumer connected\n");
540  }
541  }

544  ACE_PRINT_EXCEPTION (sysex,
545  "System Exception connecting Push Consumer :");
546  ACE_RE_THROW;
547  }
548  ACE_CATCHANY

551  "Caught exception connecting Push Consumer :");

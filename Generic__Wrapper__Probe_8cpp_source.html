331  raw_data.append(buff);
332 
333  string::size_type index = raw_data.find("\n");
334  while ( index != string::npos )
335  {
336  command.assign(raw_data, 0,

338  raw_data.erase(0, index + strlen("\n"));
339  index = raw_data.find("\n");

341  }
342  }

344 
345  DEBUG0 (2, "PROBE Listener: Leaving the Listener thread\n");
346  return 0;

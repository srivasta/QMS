398  assert(lookup_p);

400 
401  // Perparing to do an import operation. Should some of these go
402  // into the sensor object itself?
403  PolicySeq policies; //empty sequence
404 
405  CosTrading::Lookup::SpecifiedProps desired_props; // No properties

434  } /* end of if(offer_seq_out->length() == 0) */

128 {
129  QMS_Trace("internal_deploy", __LINE__,__FILE__);

131 
132  DEBUG0 (4, "Continuous_Pipe_Proxy_Remos:: Forking a child");
133  int childpid = fork();
134  if(childpid < 0)
135  {
136  perror("Fork Failed");
137  return childpid;
138  } /* end of if(childpid < 0) */
139 
140  if(childpid == 0)
141  { // in the child
142 
143  DEBUG0 (4, "DEBUG: Handling descriptors in the child");
144  if (close(ctrlpipe[1]) != 0)
145  perror ("close");
146  if (close(datapipe[0]) != 0)
147  perror ("close");
148 
149  DEBUG0 (2, "Continuous_Pipe_Proxy_Remos:: Handling stdin in the child");
150  if (ctrlpipe[0] != STDIN_FILENO ) {
151  if ( dup2 ( ctrlpipe[0], STDIN_FILENO ) != STDIN_FILENO ) {
152  fprintf(stderr, "Problem with STDIN dup2\n");
153  return (-1);
154  }
155  if (close ( ctrlpipe[0]) != 0)
156  perror("close");
157  }
158 
159  DEBUG0 (2, "Continuous_Pipe_Proxy_Remos:: Handling stdout in the child");
160  if ( datapipe[1] != STDOUT_FILENO ) {
161  if ( dup2 ( datapipe[1], STDOUT_FILENO ) != STDOUT_FILENO ) {
162  // Problems
163  fprintf(stderr, "Problem with STDOUT dup2\n");
164  return (-1);
165  }
166  if (close ( datapipe[1]) != 0)
167  perror("close datapipe[0]");
168  }
169 
170  DEBUG1 (2, "DEBUG: Continuous_Pipe_Proxy_Remos::deploy exec %s",
171  file_name_p_);
172  // DEBUG!!!
173  if(!argv_)
174  {
175  fprintf(stderr, "No arguments!!!\n");
176  return (-1);
177  } /* end of if(!this->argv_) */
178  else
179  {
180  int i = 0;
181  std::cerr << "DEBUG: Command line arguments: ";
182  while(argv_[i])
183  {
184  std::cerr << argv_[i++] << " ";
185  } /* end of while(this->argv_[i]) */
186  std::cerr << std::endl;
187  } /* end of else */
188 
189  int ret = execvp(file_name_p_, argv_);
190  perror("Failed execve");
191  DEBUG1 (8, "DEBUG: Failed execve %d", ret);
192  } /* end of if(int childpid = fork()) */
193  else
194  { // in the parent
195  DEBUG0 (4, "DEBUG: Handling descriptors in parent");
196  if (close(ctrlpipe[0]) != 0)
197  perror ("close");
198  if (close(datapipe[1]) != 0)
199  perror ("close");
200 
201  char buff[BUFSIZ];
202  ssize_t length ;
203  ssize_t n ;
204 
205  DEBUG0 (6, "DEBUG: In parent: writing start");
206 
207  strcpy (buff, "start\n");
208  length = strlen(buff);
209 
210  ssize_t nleft = length;
211  ssize_t nwritten = 0;
212  char *ptr = buff;
213 
214  while(nleft > 0)
215  {
216  if((nwritten = write(ctrlpipe[1], ptr, nleft)) <= 0)
217  {
218  if(errno == EINTR)
219  {
220  nwritten = 0;
221  } /* end of if(errno == EINTR) */
222  else
223  {
224  perror ("Error while writing start on to pipe");
225  } /* end of else */
226  } /* end of if((nwritten = write(sockfd, ptr, nleft)) <= 0) */
227  nleft -= nwritten;
228  ptr += nwritten;
229  } /* end of while(nleft > 0) */
230 
231  string message;
232  string raw_data;
233  bool done = false;
234 
235  DEBUG0 (4, "DEBUG: In parent: reading");
236  while(!done)
237  {
238  n = read(datapipe[0], buff, BUFSIZ);
239  if(n < 0)
240  {
241  if(errno == EINTR)
242  {
243  continue;
244  } /* end of if(errno == EINTR) */
245  else
246  {
247  break;
248  } /* end of else */
249  } /* end of if(n < 0) */
250  if(n == 0)
251  {
252  break;
253  } /* end of if(n == 0) */
254 
255  buff[n] = 0;
256  DEBUG2 (2, "DEBUG: Read %d bytes --> \n%s", n, buff);
257  raw_data.append(buff);
258 
259  string::size_type index =
260  raw_data.find(STDC::QMS::Citizen::delimiter);
261  while(index != string::npos)
262  {
263  // we have at least one complete message in here
264  message.assign(raw_data, 0, index);
265  raw_data.erase(0, index + strlen(STDC::QMS::Citizen::delimiter));
266  index = raw_data.find(STDC::QMS::Citizen::delimiter);
267  _metadata.handle_data(_key, message.c_str());
268  } /* end of while(message.find(delimiter) != string::npos) */
269  } /* end of while((n = read(datapipe[0], buff, BUFSIZ) ) > 0)*/
270  if(n == -1)
271  {
272  perror("read from child\n");
273  } /* end of if(n == -1) */
274 
275  DEBUG0 (6, "DEBUG: In parent: finished reading");
276 
277  int status = 0;
278  pid_t pid_ret;
279  while((pid_ret = waitpid(childpid, &status, 0) ) != childpid)
280  {
281  DEBUG1(4, "DEBUG: Exited Childpid ->%d", pid_ret);
282  } /* end of while(pid_ret = waitpid(childpid, &status, 0)) */
283  if(WIFEXITED(status))
284  {
285  DEBUG0(4, "DEBUG: Child Exited Normally");
286  } /* end of if(WIFEXITED(status)) */
287  else
288  {
289  DEBUG1(6, "DEBUG: Child Exited Abnormally with status %d",
290  WEXITSTATUS(status));
291  if(WIFSIGNALED(status))
292  {
293  DEBUG1(6, "DEBUG: Child got signal %d",
294  WTERMSIG(status));
295  } /* end of if(WIFSIGNALED(status)) */
296  } /* end of else if(WIFEXITED(status)) */
297  DEBUG0 (4, "DEBUG: Parent closing file descriptors");
298  if (close(ctrlpipe[1]) != 0)
299  perror ("close");
300  if (close(datapipe[0]) != 0)
301  perror ("close");
302 
303  } /* end of else */
304  return (0);
305 }

336

343  perror ("Error while creating the datapipe");

345  }
346

362  } /* end of if(pipe(this->datapipe) < 0) */
363 #endif

365  }

367

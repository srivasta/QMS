234  assert(lookup_p);

236 
237  // Perparing to do an import operation. Should some of these go
238  // into the sensor object itself?
239  PolicySeq policies; //empty sequence
240 
241  CosTrading::Lookup::SpecifiedProps desired_props; // No properties

330  QMS_Trace("Test_Receiver::choose_offer", __LINE__, __FILE__);

333  int i = 0;
334 
335  const CosTrading::OfferSeq& offer_seq = *offer_seq_ptr;
336  for (length = offer_seq.length (), i = 0; i < length; i++)
337  {
338  // Call back to the exported object.
339  STDC::QMS::Citizen::Command_var sensor_var =
340  STDC::QMS::Citizen::Command::_narrow(offer_seq[i].reference.in());
341  CORBA::Short acknowledgement = 0;
342  DEBUG1(2, "DEBUG: Looking at offer %d\n", i);
343  if(CORBA::is_nil(sensor_var.in()))
344  {
345  cerr << "The reference is not a QMS Citizen" << endl
346  << "For offer in sequence " << i << endl;
347  continue;
348  } /* end of if(CORBA::is_nil(hello_var.in())) */
349  sensor_var->ping(acknowledgement);
350  if(acknowledgement == 0)
351  {
352  cerr << "Could not ping Sensor" << endl
353  << "For offer " << i << endl;
354  continue;
355  } /* end of if(acknowledgement == 0) */
356  this->handle_selected_properties(offer_seq[i].properties,
357  ACE_TRY_ENV);
358  cerr << "ping succeded\n";
359  return;
360  }
361 
362  // Well, the sequence failed us. Now we need to look at the iterator
363  if(!CORBA::is_nil(iterator.in()))
364  {
365  DEBUG0(2, "DEBUG: The iterator is not null\n");
366  bool more;
367  do
368  {
369  CORBA::ULong length = 10;
370  CosTrading::OfferSeq *iter_offers_ptr;
371  CosTrading::OfferSeq_out iter_offers_out(iter_offers_ptr);
372 
373  more = iterator->next_n(length, iter_offers_out);
374 
375  CosTrading::OfferSeq_var offers (iter_offers_ptr);
376 
377  for(CORBA::ULong i = 0; i < offers->length(); i++)
378  {
379  CORBA::Short acknowledgement = 0;
380  DEBUG1(2, "DEBUG: Looking at offer %d\n", i);
381  STDC::QMS::Citizen::Command_var sensor_var =
382  STDC::QMS::Citizen::Command::_narrow(offers[i].reference.in());
383  if(CORBA::is_nil(sensor_var.in()))
384  {
385  cerr << "The reference is not a QMS Citizen" << endl
386  << "For offer in iterator " << i << endl;
387  continue;
388  } /* end of if(CORBA::is_nil(hello_var.in())) */
389  sensor_var->ping(acknowledgement);
390  if(acknowledgement == 0)
391  {
392  cerr << "Could not ping Sensor" << endl
393  << "For offer " << i << endl;
394  continue;
395  } /* end of if(acknowledgement == 0) */
396 
397  this->handle_selected_properties(offers[i].properties,
398  ACE_TRY_ENV);
399  DEBUG0(4, "DEBUG: ping succeded\n");
400  return;
401  } /* end of for(i = 0; i < offers->length(); i++) */
402  }
403  while(more);
404  } /* end of if(!CORBA::is_nil(iterator.in())) */
405  else

407  DEBUG0(2, "DEBUG: The iterator is null\n");

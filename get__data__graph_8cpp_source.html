187 StringToCharHosts(string hostnames, char *node_names[])
188 {
189  string x;
190  int len, i = 0;
191  int num_nodes = 0;
192 
193  if (!hostnames.empty())
194  {
195  bool done = false;
196  size_t location = 0;
197  size_t start = 0;
198  size_t stop = hostnames.length();
199  while (done == false)
200  {
201  location = hostnames.find(",", start);
202  if (location == string::npos)
203  {
204  x.assign(hostnames.substr(start, hostnames.length()-1));
205  len = x.length();
206  node_names[i] = new char [len + 1];
207  x.copy(node_names[i], len, 0);
208  node_names[i][len] = 0;
209  //cout << "node " << node_names[i] << endl;
210  i++;
211 
212  done = true;
213  }
214  else
215  {
216  stop = location;
217  x.assign(hostnames.substr(start, stop - start));
218  len = x.length();
219  node_names[i] = new char [len + 1];
220  x.copy(node_names[i], len, 0);
221  node_names[i][len] = 0;
222  //cout << "node " << node_names[i] << endl;
223  i++;
224 
225  start = location +1;
226  }
227  }
228  num_nodes = i;
229  //cout << "num_nodes -> " << num_nodes << endl;
230  }
231  return num_nodes;

427 listener (void *args)

429  char buff[LENGTH];

431  string command,raw_data;
432  raw_data.erase();
433  command.erase();
434  bool done = false;
435 
436  for (;;)

438  while (!done)

441  if(n < 0)
442  {
443  if(errno == EINTR)
444  {
445  continue;
446  } /* end of if(errno == EINTR) */
447  else
448  {
449  break;
450  } /* end of else */

453  {
454  break;
455  } /* end of if(n == 0) */
456

459  raw_data.append(buff);

469  }
470  }

472 
473  DEBUG0 (2, "PROBE Listener: Leaving the Listener thread\n");
474  return 0;
